Usage

<!-- Spinner -->
<loading-spinner aria-label="Loading..."></loading-spinner>

<!-- Spinner sizes -->
<loading-spinner data-size="small"></loading-spinner>
<loading-spinner data-size="large"></loading-spinner>

<!-- Progress bar -->
<progress-bar>
  <label>Uploading...</label>
  <progress value="75" max="100">75%</progress>
  <span data-value>75%</span>
</progress-bar>

<!-- Progress with color -->
<progress-bar data-color="success">
  <progress value="100" max="100">100%</progress>
</progress-bar>
