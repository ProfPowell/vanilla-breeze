Usage

Import the component stylesheet and use the documented markup patterns.

<!-- Include component styles -->
<link rel="stylesheet" href="components.css" />

<!-- Use data attributes for variants -->
<button data-button="primary">Save</button>
<button data-button="secondary">Cancel</button>

<!-- Custom elements for semantic components -->
<status-badge data-status="success">Active</status-badge>
<user-avatar data-size="large">JD</user-avatar>

<!-- Form fields with validation -->
<form-field>
  <label for="email">Email</label>
  <input type="email" id="email" required />
</form-field>
