audio-player

retro bundle · VB component contract v1.0.0 · token layer auto-swaps with theme

Token theme:
data-visualizer="wave" (default)
data-visualizer="bars"
data-visualizer="circle"
Custom --component-* token override (gold accent)

Component token API — override anything inline

The retro chrome stays retro. Only the token layer swaps. Use --audioplayer-* custom properties to override specific aspects inline without touching the component internals. Use ::part() to restyle structural regions. 

<!-- Basic usage -->
<audio-player src="track.mp3"></audio-player>

<!-- With options -->
<audio-player
  src="track.mp3"
  data-title="My Track"
  data-visualizer="bars"
  autoplay loop>
  <!-- JS-off fallback -->
  <audio controls src="track.mp3"></audio>
</audio-player>

<!-- Custom token overrides -->
<audio-player style="
  --audioplayer-screen-color: oklch(72% 0.22 75);
  --audioplayer-bg: oklch(8% 0.015 60);
"></audio-player>

<!-- Part styling from outside -->
<style>
  audio-player::part(screen)   { border-radius: 50%; aspect-ratio: 1; }
  audio-player::part(play-button) { border-radius: 50%; }
</style>

<!-- Events -->
<script>
  document.querySelector('audio-player')
    .addEventListener('vb:audioplayer:play', e => {
      console.log('Playing', e.detail)
    })
</script>
